    def on_finish(*params)
        pid = params[0]
        code = @do_on_finish[pid] || @do_on_finish[0] or return 0
        begin
            my_argc = code.arity - 1
            if my_argc > 0
                my_params = params[0 .. my_argc]
            else
                my_params = [params[0]]
            end
            params = my_params
            code.call(*params)
        rescue => e
            raise "on finish failed: #{e.message}!\n"
        end
    end
